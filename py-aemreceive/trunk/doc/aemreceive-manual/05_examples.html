5. Examples

Event handler

def striptext(text, removing=None, fromend=AEEnum('Both')):
    method = {'Left': text.lstrip,
              'Rght': text.rstrip,
              'Both': text.strip}[fromend.code]
    return method(removing)

installeventhandler(
        striptext,
        'TeCoStrp',
        ('----', 'text',     kae.typeUnicodeText),
        ('Remo', 'removing', kae.typeUnicodeText),
        ('From', 'fromend',  ArgEnum('Left', 'Rght', 'Both'))
        )

Coercion handler

codecs = Codecs()

def aetypetounicode(desc, totype):
    s = u'\u00abclass %s\u00bb' % codecs.unpack(desc).code
    return codecs.pack(s)

installcoercionhandler(
        aetypetounicode, 
        kae.typeType, 
        kae.typeUnicodeText)
